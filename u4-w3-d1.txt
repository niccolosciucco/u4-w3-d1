1. SELECT * FROM clienti
WHERE nome = 'Mario'

2. SELECT nome, cognome FROM clienti
WHERE anno_di_nascita = 1982

3. SELECT COUNT(*)
FROM fatture
WHERE iva = 20;

4. SELECT * FROM prodotti 
WHERE EXTRACT(YEAR FROM data_attivazione) = 2017 
AND (in_produzione = true OR in_commercio = true);

5. SELECT * FROM fatture 
JOIN clienti on fatture.id_cliente = clienti.numero_cliente
WHERE importo < 1000

6. SELECT numero_fattura, importo, iva, data_fattura, fornitori.denominazione
FROM fatture 
JOIN fornitori on numero_fornitore = id_fornitore

7. SELECT EXTRACT(YEAR FROM data_fattura), COUNT(*) 
FROM fatture 
WHERE iva = 20
GROUP BY EXTRACT(YEAR FROM data_fattura)
